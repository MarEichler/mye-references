2.1.1 {waldo}

waldo::compare(c("a", "b", "c"), c("a", "b"))
`old`: "a" "b" "c"
`new`: "a" "b"    
df1 <- data.frame(x = 1:3, y = 3:1)
df2 <- tibble::tibble(rev(df1))
waldo::compare(df1, df2)
`class(old)`: "data.frame"                   
`class(new)`: "tbl_df"     "tbl" "data.frame"

`names(old)`: "x" "y"
`names(new)`: "y" "x"
